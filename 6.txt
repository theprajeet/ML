import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.datasets import fetch_california_housing

california_housing = fetch_california_housing(as_frame=True)
data = california_housing.data
data['MedHouseVal'] = california_housing.target

X = data[['AveRooms']].values
y = data['MedHouseVal'].values

def locally_weighted_regression(X, y, tau):
    m = len(X)
    y_pred = np.zeros(m)
    for i in range(m):
        weights = np.exp(-np.sum((X - X[i])**2, axis=1) / (2 * tau**2))
        W = np.diag(weights)
        try:
            theta = np.linalg.inv(X.T @ W @ X) @ X.T @ W @ y
            y_pred[i] = X[i] @ theta
        except np.linalg.LinAlgError:
            y_pred[i] = np.mean(y)
    return y_pred

X_normalized = (X - np.mean(X, axis=0)) / np.std(X, axis=0)
X_normalized = np.c_[np.ones(X_normalized.shape), X_normalized]

np.random.seed(0)
subset_indices = np.random.choice(len(X_normalized), size=500, replace=False)
X_subset = X_normalized[subset_indices]
y_subset = y[subset_indices]

tau = 0.5
y_pred_subset = locally_weighted_regression(X_subset, y_subset, tau)

plt.figure(figsize=(10, 6))
plt.scatter(X_subset[:, 1], y_subset, color='blue', alpha=0.5, label='Data')
plt.plot(np.sort(X_subset[:, 1]), np.sort(y_pred_subset), color='red', linewidth=2, label='LWR Fit')
plt.xlabel('Average Rooms')
plt.ylabel('Median House Value')
plt.title('California Housing: Locally Weighted Regression')
plt.legend()
plt.show()
